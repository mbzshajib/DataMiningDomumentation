\begin{equation}
\emph ExpSup \qquad = \qquad \sum_{i = 0}^{UDB} [\prod_{x \in I } p(x , t_i)]
\end{equation}
\begin{center}


\textbf{\emph {where,}}\\ 
\begin{itemize}
\item
\textbf{\emph {I}} is itemset,

\item
\textbf{\emph {UDB}} is an uncertain database.

\end{itemize}
